val rdd = sc.textFile("file:///home/hadoop/Desktop/Sparkcount.txt")

rdd.collect()

val rdd1 = rdd.flatMap(x => x.split(" "))

rdd1.collect()

val rdd2 = rdd1.map(x => (x,1))

rdd2.collect()

val rdd3 = rdd2.reduceByKey((a,b) => a+b)

rdd3.collect()

rdd3.toDF("word","count").show()
